```python
import pytest
from solvers.linear_solver import LinearSolver

def test_linear_solver_solve(linear_solver):
    A = np.array([[1, 2], [3, 4]])
    b = np.array([5, 6])
    x = linear_solver.solve(A, b)
    assert np.allclose(x, np.linalg.solve(A, b))

def test_linear_solver_inverse(linear_solver):
    A = np.array([[1, 2], [3, 4]])
    inverse = linear_solver.inverse(A)
    assert np.allclose(inverse, np.linalg.inv(A))
```
